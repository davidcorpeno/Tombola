    const display = document.getElementById("nameDisplay");
    let currentNames = [];

    onValue(namesRef, (snapshot) => {
      currentNames = snapshot.val() || [];
      display.innerText = "Nombres restantes: " + currentNames.length;
    });

    // Cargar nombres en Firebase
    document.getElementById('loadBtn').addEventListener('click', async ()=>{
      const text = document.getElementById('nameInput').value.trim();
      const list = text.split(/\n+/).map(s=>s.trim()).filter(s=>s);
      if(list.length===0){ alert('Escribe al menos 1 nombre.'); return; }
      await set(namesRef, list);
      alert('Nombres cargados online: '+list.length);
    });

    // Sacar nombre aleatorio y eliminarlo de Firebase
    document.getElementById('drawBtn').addEventListener('click', async ()=>{
      if(currentNames.length===0){ alert('No hay nombres disponibles.'); return; }
      const i = Math.floor(Math.random()*currentNames.length);
      const picked = currentNames.splice(i,1)[0];
      await set(namesRef, currentNames);
      alert('¡Tu persona: ' + picked + '!');
    });
